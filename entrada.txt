function boolean esPrimo(integer numero)
var integer contador := 2;
var boolean primo := true;

begin
	while((primo) && (contador != numero))
	{
		if (numero % contador == 0)
		{

			primo := false;
		}

		contador++;
	}
	
	return primo;

end
